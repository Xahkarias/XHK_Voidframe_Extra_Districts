# ktisma
building_xhk_extraframe_framebirch_ktisma_foundry = {
    base_buildtime = @b2_time
    category = manufacturing
    empire_limit = 1

    building_sets = {
        xhk_extraframe_ktisma
    }

    resources = {
        category = planet_buildings
        cost = {
            minerals = @b2_minerals
        }
    }

    potential = {
        is_planet_class = pc_giga_frameworld
        has_planet_flag = xhk_extraframe_framebirch
    }

    allow = { has_major_upgraded_capital = yes }

    planet_modifier = {
        planet_metallurgists_produces_mult = 0.05
    }
    triggered_planet_modifier = {
        potential = { exists = owner owner = { is_regular_birch_empire = yes } }
        planet_metallurgists_consumer_goods_upkeep_add = 1.5
    }
    triggered_planet_modifier = {
        potential = { exists = owner owner = { is_gestalt = yes } }
        planet_metallurgists_energy_upkeep_add = 3
    }
}
