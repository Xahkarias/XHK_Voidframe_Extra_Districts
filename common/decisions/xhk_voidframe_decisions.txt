				}
			}
		}
	}
	
	effect = {
		custom_tooltip = xhk_voidframe_add_hyperstructural_effect

		hidden_effect = {
			set_planet_flag = xhk_voidframe_hyperstructural

			from = {
				change_variable = { which = giga_current_hyperstructural_ass value = 1 }
