xhk_extraframe_orbring_shipyard = {
	icon = "GFX_starbase_shipyard"
	section = "XHK_EXTRAFRAME_SHIPYARD_ORBITAL_RING_SECTION"
	construction_days = 180

	potential = {
		has_starbase_size = xhk_extraframe_orbring
		solar_system = {
			count_starbase_modules = {
				type = xhk_extraframe_orbring_hatchery
				count < 1
				include_being_constructed = yes
			}
			count_starbase_modules = {
				type = xhk_extraframe_orbring_beastport
				count < 1
				include_being_constructed = yes
			}
		}
	}

	possible = {
		solar_system = {
			count_starbase_modules = {
				type = xhk_extraframe_orbring_hatchery
				count < 1
				include_being_constructed = yes
			}
			count_starbase_modules = {
				type = xhk_extraframe_orbring_beastport
				count < 1
				include_being_constructed = yes
			}
		}
	}

	abort_construction_trigger = {
		OR = {
			has_starbase_module = xhk_extraframe_orbring_hatchery
			has_starbase_module = xhk_extraframe_orbring_beastport
		}
	}

	resources = {
		category = starbase_modules
		cost = {
			alloys = 2000
		}
		upkeep = {
			energy = 10
		}
	}

	station_modifier = {
		starbase_shipyard_capacity_add = 3
	}

	on_finished = {
		set_ship_construction_type = starbase_shipyard
	}
	on_destroyed = {
		if = {
			limit = { NOT = { has_starbase_module = xhk_extraframe_orbring_shipyard } }
			set_ship_construction_type = none
		}
	}

	ai_weight = {
		weight = 0
	}
}

xhk_extraframe_orbring_hatchery = {
	icon = "GFX_starbase_hatchery"
	section = "XHK_EXTRAFRAME_SHIPYARD_ORBITAL_RING_SECTION"
	construction_days = 180

	potential = {
		exists = owner
		has_starbase_size = xhk_extraframe_orbring
		owner = {
			has_technology = tech_alien_cloning
			is_beastmasters_empire = no
		}
		solar_system = {
			count_starbase_modules = {
				type = xhk_extraframe_orbring_shipyard
				count < 1
				include_being_constructed = yes
			}
			count_starbase_modules = {
				type = xhk_extraframe_orbring_beastport
				count < 1
				include_being_constructed = yes
			}
		}
	}

	possible = {
		solar_system = {
			count_starbase_modules = {
				type = xhk_extraframe_orbring_shipyard
				count < 1
				include_being_constructed = yes
			}
			count_starbase_modules = {
				type = xhk_extraframe_orbring_beastport
				count < 1
				include_being_constructed = yes
			}
		}
	}

	abort_construction_trigger = {
		OR = {
			has_starbase_module = xhk_extraframe_orbring_shipyard
			has_starbase_module = xhk_extraframe_orbring_beastport
		}
	}

	resources = {
		category = starbase_modules
		cost = {
			alloys = 2000
		}
		upkeep = {
			energy = 10
		}
	}

	station_modifier = {
		custom_tooltip = xhk_hatchery_ring_tooltip
		show_only_custom_tooltip = yes
		starbase_shipyard_capacity_add = 3
	}

	country_modifier = {
		country_vivarium_max_capacity_add = 5
	}

	on_finished = {
		set_ship_construction_type = starbase_hatchery
	}
	on_destroyed = {
		if = {
			limit = { NOT = { has_starbase_module = xhk_extraframe_orbring_hatchery } }
			set_ship_construction_type = none
		}
	}

	triggered_station_modifier = {
		potential = {
			exists = planet
			planet = {
				has_deposit = xhk_extraframe_vivarium
			}
		}
		custom_tooltip = xhk_hatchery_ring_tooltip_more
		show_only_custom_tooltip = yes
		starbase_shipyard_capacity_add = 7
	}

	ai_weight = {
		weight = 0
	}
}
