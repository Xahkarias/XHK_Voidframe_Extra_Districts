
### eludium tech stuff
pd_can_have_floating_tech = {
	OR = {
		any_owned_planet = {
			has_modifier = pd_floating
		}
		capital_scope = {
			has_deposit = xhk_extraframe_eludium
		}
	}
}

pd_can_have_crystal_tech = {
	OR = {
		any_owned_planet = {
			has_modifier = pd_crystal
		}
		capital_scope = {
			OR = {
				has_deposit = xhk_extraframe_crystal
				has_deposit = xhk_extraframe_crystal_activated
			}
		}
	}
}
