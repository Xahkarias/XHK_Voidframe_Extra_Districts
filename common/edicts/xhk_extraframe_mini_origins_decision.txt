
# override order for knights a bit of a pain in the ass
# Decision origins gets first dibs
# then extra goodies overwrites
xhk_extraframe_decision_origins_knightframe = {
	length = 1
	icon = "GFX_edict_type_policy"
	
	resources = {
		category = edicts
	}

	potential = {
		has_toxoids = yes
		is_ai = no
		is_gestalt = no
		giga_has_frameworld_origin = yes
		days_passed < 180
		NOT = {
			has_country_flag = decision_toxic_knights
		}
		xhk_decision_origins_mod_exists = yes
	}

	effect = {
		# capital stuff
		hidden_effect = {
			set_country_flag = decision_toxic_knights
			capital_scope = {			
				set_planet_flag = toxic_knights_habitat

				change_planet_size = 5

				add_deposit = d_toxic_god_blight_upon_the_land
				add_deposit = d_toxic_god_deitys_swarms
				add_deposit = d_toxic_god_pools_most_venemous
				add_deposit = d_toxic_god_pestilential_wasteland
				add_deposit = d_toxic_god_envenomed_seas
				add_deposit = d_knights_calibrator
				
				add_district = district_xhk_extraframe_knights
				add_building = {
					district = district_xhk_extraframe_knights
					zone = zone_xhk_extraframe_knight
					building = building_order_keep
				}			
				
			}
		}
		add_research_option = tech_habitat_1

		if = {
			limit = {
				OR = {
					is_lithoid_empire = no
					is_catalytic_empire = yes
				}
			}
			give_technology = {
				tech = tech_eco_simulation
				message = no
			}
			give_technology = {
				tech = tech_hydroponics
				message = no
			}
		}
		else = {
			give_technology = {
				tech = tech_powered_exoskeletons
				message = no
			}
			give_technology = {
				tech = tech_space_mining_1
				message = no
			}
		}
		
		# mimics toxoids.7000
		start_situation = {
			type = situation_toxic_knights
		}
		custom_tooltip = "edict_decision_origins_knightframe_tt"
		#show_only_custom_tooltip = no
	}
}

xhk_extraframe_decision_origins_necroframe = {
    length = 1
    icon = "GFX_edict_type_policy"

    resources = {
        category = edicts
    }

    potential = {
        has_necroids = yes
        is_ai = no
        giga_has_frameworld_origin = yes
        days_passed < 180
        NOR = {
            has_necrophage_origin = yes
            is_machine_empire = yes
            is_individual_machine = yes
        }
        xhk_decision_origins_mod_exists = yes
    }

    effect = {
        hidden_effect = {
            set_country_flag = decision_origins_necrophage
            country_event = { id = xhk_decision_origin_triggered.15 }
            if = {
                limit = { is_hive_empire = yes }
                while = { # kill less of the starting pop
                    count = 12
                    random_owned_pop_group = {
                        kill_pop_group = {
                            POP_GROUP = this
                            AMOUNT = 100
                        }
                    }
                }
            }
            else = {
                while = { # kill half the starting pop
                    count = 24
                    random_owned_pop_group = {
                        kill_pop_group = {
                            POP_GROUP = this
                            AMOUNT = 100
                        }
                    }
                }
            }

            founder_species = {
                change_species_characteristics = {
                    add_trait = trait_necrophage
                }
            }

            capital_scope = {
                create_species = {
                    name = random
                    class = random
                    namelist = random
                    portrait = random
                    homeworld = planet
                    traits = {
                        trait = trait_rapid_breeders
                    }
                }

                # make prepatent
                if = {
                    limit = { owner = {is_hive_empire = yes } }
                    create_pop_group = {
                        size = 1200
                        species = last_created_species
                    }
                }
                else = {
                    create_pop_group = {
                        size = 2400
                        species = last_created_species
                    }
                }


                # if hive
                if = {
                    limit = { owner = { is_hive_empire = yes } }
                    last_created_species = {
                        set_citizenship_type = {
                            country = root.owner
                            type = citizenship_slavery
                        }
                        set_population_controls = {
                            country = root.owner
                            type = population_control_no
                        }
                    }
                }

                add_district = district_xhk_extraframe_necrophage
                add_building = {
                    district = district_xhk_extraframe_necrophage
                    zone = zone_xhk_extraframe_necrophage
                    building = building_necrophage_elevation_chamber
                }
            }

        }
        custom_tooltip = "edict_decision_origins_necrophage_desc_tt"

    }
}
